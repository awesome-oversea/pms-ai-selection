### 2.1 OAuth2认证流程

```python
from authlib.integrations.fastapi_client import OAuth
from fastapi import FastAPI, Request, Depends
from fastapi.security import OAuth2AuthorizationCodeBearer

class OAuth2Config:
    AUTHORIZATION_URL = "https://auth.fms.example.com/oauth/authorize"
    TOKEN_URL = "https://auth.fms.example.com/oauth/token"
    USERINFO_URL = "https://auth.fms.example.com/oauth/userinfo"
    CLIENT_ID = "fms-client"
    CLIENT_SECRET = "${OAUTH_CLIENT_SECRET}"
    REDIRECT_URI = "https://fms.example.com/callback"

oauth2_scheme = OAuth2AuthorizationCodeBearer(
    authorizationUrl=OAuth2Config.AUTHORIZATION_URL,
    tokenUrl=OAuth2Config.TOKEN_URL
)

class AuthService:
    def __init__(self):
        self.oauth = OAuth()
        self.oauth.register(
            name="fms",
            client_id=OAuth2Config.CLIENT_ID,
            client_secret=OAuth2Config.CLIENT_SECRET,
            authorize_url=OAuth2Config.AUTHORIZATION_URL,
            token_url=OAuth2Config.TOKEN_URL,
            userinfo_url=OAuth2Config.USERINFO_URL
        )
    
    async def authenticate(self, token: str) -> dict:
        user_info = await self.oauth.fms.userinfo(token)
        return user_info
    
    async def login(self, request: Request):
        redirect_uri = OAuth2Config.REDIRECT_URI
        return await self.oauth.fms.authorize_redirect(request, redirect_uri)
    
    async def callback(self, request: Request):
        token = await self.oauth.fms.authorize_access_token(request)
        user_info = await self.oauth.fms.userinfo(token)
        return user_info
```

### 2.2 JWT Token管理

```python
from datetime import datetime, timedelta
from jose import jwt, JWTError
from pydantic import BaseModel

class TokenPayload(BaseModel):
    sub: str
    exp: datetime
    iat: datetime
    roles: list[str]
    tenant_id: str

class JWTManager:
    SECRET_KEY = "${JWT_SECRET_KEY}"
    ALGORITHM = "HS256"
    ACCESS_TOKEN_EXPIRE_MINUTES = 30
    REFRESH_TOKEN_EXPIRE_DAYS = 7
    
    @classmethod
    def create_access_token(
        cls,
        user_id: str,
        roles: list[str],
        tenant_id: str
    ) -> str:
        expire = datetime.utcnow() + timedelta(minutes=cls.ACCESS_TOKEN_EXPIRE_MINUTES)
        
        payload = {
            "sub": user_id,
            "exp": expire,
            "iat": datetime.utcnow(),
            "roles": roles,
            "tenant_id": tenant_id,
            "type": "access"
        }
        
        return jwt.encode(payload, cls.SECRET_KEY, algorithm=cls.ALGORITHM)
    
    @classmethod
    def create_refresh_token(cls, user_id: str) -> str:
        expire = datetime.utcnow() + timedelta(days=cls.REFRESH_TOKEN_EXPIRE_DAYS)
        
        payload = {
            "sub": user_id,
            "exp": expire,
            "iat": datetime.utcnow(),
            "type": "refresh"
        }
        
        return jwt.encode(payload, cls.SECRET_KEY, algorithm=cls.ALGORITHM)
    
    @classmethod
    def verify_token(cls, token: str) -> TokenPayload:
        try:
            payload = jwt.decode(token, cls.SECRET_KEY, algorithms=[cls.ALGORITHM])
            return TokenPayload(**payload)
        except JWTError:
            raise AuthenticationError("Invalid token")
    
    @classmethod
    def refresh_access_token(cls, refresh_token: str) -> str:
        payload = cls.verify_token(refresh_token)
        
        if payload.type != "refresh":
            raise AuthenticationError("Invalid refresh token")
        
        return cls.create_access_token(
            user_id=payload.sub,
            roles=payload.roles,
            tenant_id=payload.tenant_id
        )
```

### 2.3 多因素认证(MFA)

```python
import pyotp
import qrcode
from io import BytesIO

class MFAService:
    ISSUER = "FMS-AI选品系统"
    
    def __init__(self):
        self.totp_cache = {}
    
    def generate_secret(self, user_id: str) -> str:
        secret = pyotp.random_base32()
        self.totp_cache[user_id] = secret
        return secret
    
    def get_qr_code(self, user_id: str, email: str) -> str:
        secret = self.totp_cache.get(user_id)
        if not secret:
            secret = self.generate_secret(user_id)
        
        uri = pyotp.totp.TOTP(secret).provisioning_uri(
            name=email,
            issuer_name=self.ISSUER
        )
        
        qr = qrcode.QRCode(version=1, box_size=10, border=5)
        qr.add_data(uri)
        qr.make(fit=True)
        
        img = qr.make_image(fill_color="black", back_color="white")
        
        buffer = BytesIO()
        img.save(buffer, format="PNG")
        
        return buffer.getvalue()
    
    def verify_code(self, user_id: str, code: str) -> bool:
        secret = self.totp_cache.get(user_id)
        if not secret:
            return False
        
        totp = pyotp.TOTP(secret)
        return totp.verify(code, valid_window=1)
```

## 3. 授权系统设计

### 3.1 RBAC模型

```python
from enum import Enum
from pydantic import BaseModel
from typing import Optional

class Permission(str, Enum):
    READ = "read"
    WRITE = "write"
    DELETE = "delete"
    ADMIN = "admin"

class Role(str, Enum):
    ADMIN = "admin"
    MANAGER = "manager"
    ANALYST = "analyst"
    OPERATOR = "operator"
    VIEWER = "viewer"

ROLE_PERMISSIONS = {
    Role.ADMIN: [
        Permission.READ, Permission.WRITE, Permission.DELETE, Permission.ADMIN
    ],
    Role.MANAGER: [
        Permission.READ, Permission.WRITE, Permission.DELETE
    ],
    Role.ANALYST: [
        Permission.READ, Permission.WRITE
    ],
    Role.OPERATOR: [
        Permission.READ, Permission.WRITE
    ],
    Role.VIEWER: [
        Permission.READ
    ]
}

class User(BaseModel):
    user_id: str
    username: str
    email: str
    roles: list[Role]
    tenant_id: str
    is_active: bool = True

class RBACService:
    def __init__(self, db: Database):
        self.db = db
    
    async def get_user(self, user_id: str) -> Optional[User]:
        user_data = await self.db.get("users", user_id)
        if user_data:
            return User(**user_data)
        return None
    
    def has_permission(self, user: User, permission: Permission) -> bool:
        for role in user.roles:
            if permission in ROLE_PERMISSIONS.get(role, []):
                return True
        return False
    
    async def check_access(
        self,
        user_id: str,
        resource: str,
        action: Permission
    ) -> bool:
        user = await self.get_user(user_id)
        
        if not user or not user.is_active:
            return False
        
        return self.has_permission(user, action)
```

```python
from functools import wraps

class ResourcePermission:
    def __init__(self, resource: str, action: Permission):
        self.resource = resource
        self.action = action

def require_permission(resource: str, action: Permission):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, current_user: User = None, **kwargs):
            if not current_user:
                raise AuthenticationError("Not authenticated")
            
            rbac = get_rbac_service()
            has_access = await rbac.check_access(
                current_user.user_id,
                resource,
                action
            )
            
            if not has_access:
                raise AuthorizationError(
                    f"No permission to {action} {resource}"
                )
            
            return await func(*args, current_user=current_user, **kwargs)
        return wrapper
    return decorator

class TenantIsolation:
    def __init__(self, db: Database):
        self.db = db
    
    async def filter_by_tenant(
        self,
        query: dict,
        tenant_id: str
    ) -> dict:
        query["tenant_id"] = tenant_id
        return query
    
    async def check_tenant_access(
        self,
        user: User,
        resource_tenant_id: str
    ) -> bool:
        return user.tenant_id == resource_tenant_id
```

```python
from enum import Enum
from datetime import datetime

class AuditAction(str, Enum):
    LOGIN = "login"
    LOGOUT = "logout"
    CREATE = "create"
    READ = "read"
    UPDATE = "update"
    DELETE = "delete"
    EXPORT = "export"
    IMPORT = "import"

class AuditLog(BaseModel):
    log_id: str
    timestamp: datetime
    user_id: str
    username: str
    tenant_id: str
    action: AuditAction
    resource_type: str
    resource_id: str
    ip_address: str
    user_agent: str
    request_method: str
    request_path: str
    request_params: dict
    response_status: int
    duration_ms: int
    status: str
    error_message: Optional[str] = None

class AuditService:
    def __init__(self, db: Database, kafka: KafkaProducer):
        self.db = db
        self.kafka = kafka
    
    async def log(
        self,
        user: User,
        action: AuditAction,
        resource_type: str,
        resource_id: str,
        request: Request,
        response_status: int,
        duration_ms: int,
        status: str = "success",
        error_message: str = None
    ):
        audit_log = AuditLog(
            log_id=str(uuid.uuid4()),
            timestamp=datetime.now(),
            user_id=user.user_id,
            username=user.username,
            tenant_id=user.tenant_id,
            action=action,
            resource_type=resource_type,
            resource_id=resource_id,
            ip_address=self._get_client_ip(request),
            user_agent=request.headers.get("user-agent", ""),
            request_method=request.method,
            request_path=str(request.url.path),
            request_params=dict(request.query_params),
            response_status=response_status,
            duration_ms=duration_ms,
            status=status,
            error_message=error_message
        )
        
        await self.db.insert("audit_logs", audit_log.dict())
        
        await self.kafka.send("audit-log-events", audit_log.dict())
    
    def _get_client_ip(self, request: Request) -> str:
        forwarded = request.headers.get("X-Forwarded-For")
        if forwarded:
            return forwarded.split(",")[0].strip()
        return request.client.host
```

### 4.2 审计中间件

```python
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware
import time

class AuditMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        start_time = time.time()
        
        response = await call_next(request)
        
        duration_ms = int((time.time() - start_time) * 1000)
        
        user = getattr(request.state, "user", None)
        
        if user and self._should_audit(request):
            audit_service = get_audit_service()
            await audit_service.log(
                user=user,
                action=self._get_action(request),
                resource_type=self._get_resource_type(request),
                resource_id=self._get_resource_id(request),
                request=request,
                response_status=response.status_code,
                duration_ms=duration_ms
            )
        
        return response
    
    def _should_audit(self, request: Request) -> bool:
        excluded_paths = ["/health", "/metrics", "/docs", "/openapi.json"]
        return not any(
            request.url.path.startswith(path) for path in excluded_paths
        )
    
    def _get_action(self, request: Request) -> AuditAction:
        method_action_map = {
            "GET": AuditAction.READ,
            "POST": AuditAction.CREATE,
            "PUT": AuditAction.UPDATE,
            "PATCH": AuditAction.UPDATE,
            "DELETE": AuditAction.DELETE
        }
        return method_action_map.get(request.method, AuditAction.READ)
    
    def _get_resource_type(self, request: Request) -> str:
        path_parts = request.url.path.split("/")
        if len(path_parts) >= 3:
            return path_parts[2]
        return "unknown"
    
    def _get_resource_id(self, request: Request) -> str:
        path_parts = request.url.path.split("/")
        if len(path_parts) >= 4:
            return path_parts[3]
        return ""
```

## 5. 数据安全设计

### 5.1 传输加密

```python
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.backends import default_backend
import base64
import os

class EncryptionService:
    ALGORITHM = "AES-256-GCM"
    KEY_LENGTH = 32
    SALT_LENGTH = 16
    NONCE_LENGTH = 12
    
    def __init__(self, master_key: bytes):
        self.master_key = master_key
    
    def encrypt(self, plaintext: str) -> dict:
        from cryptography.hazmat.primitives.ciphers.aead import AESGCM
        
        salt = os.urandom(self.SALT_LENGTH)
        key = self._derive_key(salt)
        
        nonce = os.urandom(self.NONCE_LENGTH)
        aesgcm = AESGCM(key)
        
        ciphertext = aesgcm.encrypt(nonce, plaintext.encode(), None)
        
        return {
            "salt": base64.b64encode(salt).decode(),
            "nonce": base64.b64encode(nonce).decode(),
            "ciphertext": base64.b64encode(ciphertext).decode()
        }
    
    def decrypt(self, encrypted_data: dict) -> str:
        from cryptography.hazmat.primitives.ciphers.aead import AESGCM
        
        salt = base64.b64decode(encrypted_data["salt"])
        nonce = base64.b64decode(encrypted_data["nonce"])
        ciphertext = base64.b64decode(encrypted_data["ciphertext"])
        
        key = self._derive_key(salt)
        aesgcm = AESGCM(key)
        
        plaintext = aesgcm.decrypt(nonce, ciphertext, None)
        return plaintext.decode()
    
    def _derive_key(self, salt: bytes) -> bytes:
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=self.KEY_LENGTH,
            salt=salt,
            iterations=100000,
            backend=default_backend()
        )
        return kdf.derive(self.master_key)
```

### 5.2 数据脱敏

```python
import re

class DataMasker:
    PATTERNS = {
        "email": (r"([^@]{2})[^@]*(@[^@]*)", r"\1***\2"),
        "phone": (r"(\d{3})\d{4}(\d{4})", r"\1****\2"),
        "id_card": (r"(\d{4})\d{10}(\d{4})", r"\1**********\2"),
        "bank_card": (r"(\d{4})\d{8}(\d{4})", r"\1********\2"),
        "password": (r".*", "******")
    }
    
    @classmethod
    def mask(cls, data: str, data_type: str) -> str:
        if data_type not in cls.PATTERNS:
            return data
        
        pattern, replacement = cls.PATTERNS[data_type]
        return re.sub(pattern, replacement, data)
    
    @classmethod
    def mask_dict(cls, data: dict, sensitive_fields: list[str]) -> dict:
        masked_data = data.copy()
        
        field_type_map = {
            "email": "email",
            "phone": "phone",
            "id_card": "id_card",
            "bank_card": "bank_card",
            "password": "password"
        }
        
        for field in sensitive_fields:
            if field in masked_data:
                field_type = field_type_map.get(field, "default")
                masked_data[field] = cls.mask(
                    str(masked_data[field]),
                    field_type
                )
        
        return masked_data
```

### 6.2 安全头配置

```python
from fastapi.middleware.cors import CORSMiddleware
from fastapi.middleware.trustedhost import TrustedHostMiddleware

class SecurityMiddleware:
    @staticmethod
    def setup(app: FastAPI):
        app.add_middleware(
            TrustedHostMiddleware,
            allowed_hosts=["fms.example.com", "*.fms.example.com"]
        )
        
        app.add_middleware(
            CORSMiddleware,
            allow_origins=["https://fms.example.com"],
            allow_credentials=True,
            allow_methods=["GET", "POST", "PUT", "DELETE"],
            allow_headers=["*"],
            max_age=3600
        )
        
        @app.middleware("http")
        async def add_security_headers(request: Request, call_next):
            response = await call_next(request)
            
            response.headers["X-Content-Type-Options"] = "nosniff"
            response.headers["X-Frame-Options"] = "DENY"
            response.headers["X-XSS-Protection"] = "1; mode=block"
            response.headers["Strict-Transport-Security"] = "max-age=31536000; includeSubDomains"
            response.headers["Content-Security-Policy"] = "default-src 'self'"
            
            return response
```
