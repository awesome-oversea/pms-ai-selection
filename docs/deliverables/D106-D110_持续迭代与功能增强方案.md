### 4.2 偿还计划

```python
class TechDebtManager:
    def __init__(self):
        self.debts = []
        self.budget_ratio = 0.2
    
    def allocate_budget(self, sprint_capacity: int) -> int:
        return int(sprint_capacity * self.budget_ratio)
    
    def prioritize(self) -> list:
        return sorted(
            self.debts,
            key=lambda d: (d.priority, d.impact),
            reverse=True
        )
    
    def plan_repayment(self, sprint_capacity: int) -> list:
        budget = self.allocate_budget(sprint_capacity)
        prioritized = self.prioritize()
        
        planned = []
        remaining = budget
        
        for debt in prioritized:
            if debt.effort <= remaining:
                planned.append(debt)
                remaining -= debt.effort
        
        return planned
```

### 5.3 反馈分析

class FeedbackAnalyzer:
    def __init__(self):
        self.categories = {
            "功能建议": [],
            "体验问题": [],
            "缺陷报告": [],
            "性能问题": []
        }
    
    def analyze_trends(self, feedbacks: list) -> dict:
        trends = {
            "total": len(feedbacks),
            "by_category": {},
            "by_priority": {},
            "top_issues": []
        }
        
        for feedback in feedbacks:
            category = feedback.category
            if category not in trends["by_category"]:
                trends["by_category"][category] = 0
            trends["by_category"][category] += 1
        
        return trends
    
    def generate_report(self, period: str) -> dict:
        feedbacks = self.get_feedbacks(period)
        trends = self.analyze_trends(feedbacks)
        
        return {
            "period": period,
            "summary": trends,
            "recommendations": self._get_recommendations(trends)
        }
```

class TestAutomation:
    def __init__(self):
        self.test_suites = {
            "unit": UnitTestSuite(),
            "integration": IntegrationTestSuite(),
            "e2e": E2ETestSuite(),
            "performance": PerformanceTestSuite()
        }
    
    async def run_all(self) -> dict:
        results = {}
        
        for name, suite in self.test_suites.items():
            results[name] = await suite.run()
        
        return {
            "passed": all(r["passed"] for r in results.values()),
            "details": results
        }
```
