## 3. 多级缓存策略

### 3.1 本地缓存实现

```python
from functools import lru_cache
from cachetools import TTLCache, LRUCache
import threading

class LocalCache:
    def __init__(self, maxsize: int = 10000, ttl: int = 300):
        self.cache = TTLCache(maxsize=maxsize, ttl=ttl)
        self.lock = threading.RLock()
    
    def get(self, key: str):
        with self.lock:
            return self.cache.get(key)
    
    def set(self, key: str, value: any, ttl: int = None):
        with self.lock:
            self.cache[key] = value
    
    def delete(self, key: str):
        with self.lock:
            self.cache.pop(key, None)
    
    def clear(self):
        with self.lock:
            self.cache.clear()

class HotDataCache:
    _instance = None
    _lock = threading.Lock()
    
    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._cache = LocalCache(maxsize=5000, ttl=60)
        return cls._instance
    
    async def get_hot_products(self, category: str) -> list:
        key = f"hot_products:{category}"
        cached = self._cache.get(key)
        if cached:
            return cached
        
        products = await self._fetch_hot_products(category)
        self._cache.set(key, products)
        return products
    
    async def get_config(self, config_key: str) -> dict:
        key = f"config:{config_key}"
        cached = self._cache.get(key)
        if cached:
            return cached
        
        config = await self._fetch_config(config_key)
        self._cache.set(key, config, ttl=300)
        return config
```

### 3.2 分布式缓存实现

```python
import redis.asyncio as redis
from typing import Optional, Any
import json

class RedisCache:
    def __init__(self, redis_url: str):
        self.redis_url = redis_url
        self.client: Optional[redis.Redis] = None
    
    async def connect(self):
        self.client = await redis.from_url(
            self.redis_url,
            encoding="utf-8",
            decode_responses=True
        )
    
    async def get(self, key: str) -> Optional[Any]:
        value = await self.client.get(key)
        if value:
            return json.loads(value)
        return None
    
    async def set(
        self,
        key: str,
        value: Any,
        ttl: int = 300
    ) -> bool:
        return await self.client.setex(
            key,
            ttl,
            json.dumps(value, default=str)
        )
    
    async def delete(self, key: str) -> bool:
        return await self.client.delete(key) > 0
    
    async def mget(self, keys: list[str]) -> list[Optional[Any]]:
        values = await self.client.mget(keys)
        return [json.loads(v) if v else None for v in values]
    
    async def mset(self, mapping: dict[str, Any], ttl: int = 300) -> bool:
        pipe = self.client.pipeline()
        for key, value in mapping.items():
            pipe.setex(key, ttl, json.dumps(value, default=str))
        await pipe.execute()
        return True
    
    async def incr(self, key: str) -> int:
        return await self.client.incr(key)
    
    async def expire(self, key: str, ttl: int) -> bool:
        return await self.client.expire(key, ttl)
```

### 3.3 多级缓存协调

```python
class MultiLevelCache:
    def __init__(
        self,
        local_cache: LocalCache,
        redis_cache: RedisCache
    ):
        self.local = local_cache
        self.redis = redis_cache
    
    async def get(
        self,
        key: str,
        fetch_func: callable = None,
        ttl: int = 300
    ) -> Any:
        cached = self.local.get(key)
        if cached is not None:
            return cached
        
        cached = await self.redis.get(key)
        if cached is not None:
            self.local.set(key, cached, ttl=60)
            return cached
        
        if fetch_func:
            data = await fetch_func()
            await self.set(key, data, ttl)
            return data
        
        return None
    
    async def set(self, key: str, value: Any, ttl: int = 300):
        self.local.set(key, value, ttl=60)
        await self.redis.set(key, value, ttl)
    
    async def delete(self, key: str):
        self.local.delete(key)
        await self.redis.delete(key)
    
    async def invalidate_pattern(self, pattern: str):
        keys = await self.redis.client.keys(pattern)
        if keys:
            await self.redis.client.delete(*keys)
        
        self.local.clear()
```

## 4. 缓存更新策略

### 4.1 缓存更新模式

```python
from enum import Enum

class CacheStrategy(str, Enum):
    CACHE_ASIDE = "cache_aside"
    WRITE_THROUGH = "write_through"
    WRITE_BEHIND = "write_behind"
    REFRESH_AHEAD = "refresh_ahead"

class CacheAside:
    def __init__(self, cache: MultiLevelCache, db: Database):
        self.cache = cache
        self.db = db
    
    async def read(self, key: str, table: str, id: str) -> dict:
        cached = await self.cache.get(key)
        if cached:
            return cached
        
        data = await self.db.get(table, id)
        if data:
            await self.cache.set(key, data)
        
        return data
    
    async def write(self, key: str, table: str, id: str, data: dict):
        await self.db.update(table, {"id": id}, data)
        await self.cache.delete(key)

class WriteThrough:
    def __init__(self, cache: MultiLevelCache, db: Database):
        self.cache = cache
        self.db = db
    
    async def write(self, key: str, table: str, id: str, data: dict):
        await self.db.update(table, {"id": id}, data)
        await self.cache.set(key, data)
    
    async def read(self, key: str) -> dict:
        return await self.cache.get(key)

class RefreshAhead:
    def __init__(self, cache: MultiLevelCache, db: Database):
        self.cache = cache
        self.db = db
        self.refresh_threshold = 0.8
    
    async def read(self, key: str, table: str, id: str) -> dict:
        cached = await self.cache.get(key)
        if cached:
            ttl = await self.cache.redis.client.ttl(key)
            max_ttl = 300
            
            if ttl < max_ttl * self.refresh_threshold:
                asyncio.create_task(self._refresh(key, table, id))
            
            return cached
        
        data = await self.db.get(table, id)
        if data:
            await self.cache.set(key, data)
        
        return data
    
    async def _refresh(self, key: str, table: str, id: str):
        data = await self.db.get(table, id)
        if data:
            await self.cache.set(key, data)
```

class CacheInvalidation:
    def __init__(self, cache: MultiLevelCache, kafka: KafkaProducer):
        self.cache = cache
        self.kafka = kafka
    
    async def invalidate_on_update(
        self,
        entity_type: str,
        entity_id: str
    ):
        patterns = [
            f"{entity_type}:{entity_id}",
            f"{entity_type}:list:*",
            f"{entity_type}:search:*"
        ]
        
        for pattern in patterns:
            await self.cache.invalidate_pattern(pattern)
        
        await self.kafka.send("cache-invalidation", {
            "entity_type": entity_type,
            "entity_id": entity_id,
            "timestamp": datetime.now().isoformat()
        })
    
    async def invalidate_on_delete(
        self,
        entity_type: str,
        entity_id: str
    ):
        await self.invalidate_on_update(entity_type, entity_id)

class CacheWarmer:
    def __init__(self, cache: MultiLevelCache, db: Database):
        self.cache = cache
        self.db = db
    
    async def warmup_products(self):
        hot_products = await self.db.query(
            "products",
            {"is_hot": True},
            limit=100
        )
        
        for product in hot_products:
            key = f"product:{product['product_id']}"
            await self.cache.set(key, product, ttl=3600)

            key = f"config:{config['config_key']}"

## 5. 会话管理设计

### 5.1 会话模型

```python
from pydantic import BaseModel
from datetime import datetime
from typing import Optional

class Session(BaseModel):
    session_id: str
    user_id: str
    tenant_id: str
    
    created_at: datetime
    last_accessed: datetime
    expires_at: datetime
    
    ip_address: str
    user_agent: str
    
    data: dict = {}
    permissions: list[str] = []

class SessionManager:
    def __init__(self, redis: RedisCache):
        self.redis = redis
        self.session_ttl = 1800
    
    async def create_session(
        self,
        user_id: str,
        tenant_id: str,
        ip_address: str,
        user_agent: str
    ) -> Session:
        session_id = str(uuid.uuid4())
        now = datetime.now()
        
        session = Session(
            session_id=session_id,
            user_id=user_id,
            tenant_id=tenant_id,
            created_at=now,
            last_accessed=now,
            expires_at=now + timedelta(seconds=self.session_ttl),
            ip_address=ip_address,
            user_agent=user_agent
        )
        
        await self.redis.set(
            f"session:{session_id}",
            session.dict(),
            ttl=self.session_ttl
        )
        
        return session
    
    async def get_session(self, session_id: str) -> Optional[Session]:
        data = await self.redis.get(f"session:{session_id}")
        if data:
            return Session(**data)
        return None
    
    async def refresh_session(self, session_id: str) -> bool:
        session = await self.get_session(session_id)
        if not session:
            return False
        
        session.last_accessed = datetime.now()
        session.expires_at = datetime.now() + timedelta(seconds=self.session_ttl)
        
        await self.redis.set(
            f"session:{session_id}",
            session.dict(),
            ttl=self.session_ttl
        )
        
        return True
    
    async def destroy_session(self, session_id: str) -> bool:
        return await self.redis.delete(f"session:{session_id}")
    
    async def get_user_sessions(self, user_id: str) -> list[Session]:
        pattern = f"session:*"
        keys = await self.redis.client.keys(pattern)
        
        sessions = []
        for key in keys:
            data = await self.redis.get(key)
            if data and data.get("user_id") == user_id:
                sessions.append(Session(**data))
        
        return sessions
```

class SessionStorage:
    def __init__(self, redis: RedisCache):
        self.redis = redis
        self.local_cache = LocalCache(maxsize=1000, ttl=60)
    
    async def get(self, session_id: str) -> Optional[Session]:
        cached = self.local_cache.get(session_id)
        if cached:
            return Session(**cached)
        
        data = await self.redis.get(f"session:{session_id}")
        if data:
            self.local_cache.set(session_id, data)
            return Session(**data)
        
        return None
    
    async def set(self, session: Session):
        self.local_cache.set(session.session_id, session.dict())
        await self.redis.set(
            f"session:{session.session_id}",
            session.dict(),
            ttl=1800
        )
    
    async def delete(self, session_id: str):
        self.local_cache.delete(session_id)
        await self.redis.delete(f"session:{session_id}")

class SessionMiddleware:
    def __init__(self, session_manager: SessionManager):
        self.session_manager = session_manager
    
    async def __call__(self, request: Request, call_next):
        session_id = request.cookies.get("session_id")
        
        if session_id:
            session = await self.session_manager.get_session(session_id)
            if session:
                request.state.session = session
                await self.session_manager.refresh_session(session_id)
        
        response = await call_next(request)
        return response

class CachePreheater:
    def __init__(self, cache: MultiLevelCache, db: Database):
        self.cache = cache
        self.db = db
    
    async def preheat_all(self):
        await asyncio.gather(
            self.preheat_hot_products(),
            self.preheat_categories(),
            self.preheat_configs(),
            self.preheat_user_permissions()
        )
    
    async def preheat_hot_products(self):
        products = await self.db.query(
            "products",
            {"view_count": {"$gte": 1000}},
            limit=500
        )
        
        tasks = []
        for product in products:
            key = f"product:{product['product_id']}"
            tasks.append(self.cache.set(key, product, ttl=3600))
        
        await asyncio.gather(*tasks)
    
    async def preheat_user_permissions(self):
        users = await self.db.query("users", {"is_active": True})
        
        tasks = []
        for user in users:
            key = f"user_permissions:{user['user_id']}"
            permissions = await self._get_user_permissions(user)
            tasks.append(self.cache.set(key, permissions, ttl=1800))
        
        await asyncio.gather(*tasks)

### 6.2 缓存监控

```python
class CacheMonitor:
    def __init__(self, redis: RedisCache):
        self.redis = redis
        self.metrics = {
            "hits": 0,
            "misses": 0,
            "errors": 0
        }
    
    async def record_hit(self):
        self.metrics["hits"] += 1
    
    async def record_miss(self):
        self.metrics["misses"] += 1
    
    async def get_hit_ratio(self) -> float:
        total = self.metrics["hits"] + self.metrics["misses"]
        if total == 0:
            return 0.0
        return self.metrics["hits"] / total
    
    async def get_redis_stats(self) -> dict:
        info = await self.redis.client.info()
        return {
            "used_memory": info.get("used_memory"),
            "used_memory_peak": info.get("used_memory_peak"),
            "connected_clients": info.get("connected_clients"),
            "total_commands_processed": info.get("total_commands_processed"),
            "keyspace_hits": info.get("keyspace_hits"),
            "keyspace_misses": info.get("keyspace_misses")
        }
    
    async def get_metrics(self) -> dict:
        redis_stats = await self.get_redis_stats()
        
        return {
            "local_metrics": self.metrics,
            "hit_ratio": await self.get_hit_ratio(),
            "redis_stats": redis_stats
        }
```
