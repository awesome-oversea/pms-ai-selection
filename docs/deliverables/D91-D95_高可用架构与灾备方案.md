### 3.2 故障转移配置

```python
class DatabaseFailover:
    def __init__(self, db_config: dict):
        self.master = db_config["master"]
        self.replicas = db_config["replicas"]
        self.failover_timeout = 30
    
    async def check_master_health(self) -> bool:
        try:
            conn = await asyncpg.connect(self.master)
            await conn.execute("SELECT 1")
            await conn.close()
            return True
        except Exception:
            return False
    
    async def promote_replica(self, replica_url: str):
        conn = await asyncpg.connect(replica_url)
        await conn.execute("SELECT pg_promote()")
        await conn.close()
    
    async def update_connection_string(self, new_master: str):
        await self.config_service.update("DATABASE_URL", new_master)
        await self.notify_services()
    
    async def handle_failover(self):
        if not await self.check_master_health():
            for replica in self.replicas:
                if await self.check_replica_health(replica):
                    await self.promote_replica(replica)
                    await self.update_connection_string(replica)
                    return True
        return False
```

### 4.2 Redis Sentinel配置

```python
class RedisHA:
    def __init__(self, sentinel_hosts: list[str]):
        self.sentinel = Sentinel(
            sentinel_hosts,
            socket_timeout=5,
            socket_connect_timeout=5
        )
        self.master_name = "mymaster"
    
    async def get_master(self):
        master = self.sentinel.master_for(
            self.master_name,
            socket_timeout=5
        )
        return master
    
    async def get_replica(self):
        replica = self.sentinel.slave_for(
            self.master_name,
            socket_timeout=5
        )
        return replica
    
    async def execute_with_failover(self, operation: callable):
        try:
            master = await self.get_master()
            return await operation(master)
        except Exception:
            await asyncio.sleep(1)
            master = await self.get_master()
            return await operation(master)
```

class BackupStrategy:
    def __init__(self):
        self.backup_config = {

                "rdb_backup": "0 3 * * *",
                "aof_backup": "continuous",
                "retention_days": 7
            },
            "qdrant": {
                "snapshot": "0 4 * * *",
                "retention_days": 14
            },
            "kafka": {
                "offset_backup": "0 5 * * *",
                "retention_days": 7
            }
        }
    
    async def execute_backup(self, backup_type: str):
        if backup_type == "database":
            await self._backup_database()
        elif backup_type == "redis":
            await self._backup_redis()
        elif backup_type == "qdrant":
            await self._backup_qdrant()
    
    async def _backup_database(self):
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_file = f"/backup/postgres/fms_{timestamp}.sql"

### 5.3 灾难恢复流程

```python
class DisasterRecovery:
    def __init__(self):
        self.steps = [
            "assess_damage",
            "activate_standby",
            "restore_data",
            "verify_services",
            "switch_traffic",
            "notify_stakeholders"
        ]
    
    async def execute_recovery(self):
        results = {}
        
        for step in self.steps:
            try:
                result = await getattr(self, f"_{step}")()
                results[step] = {"status": "success", "result": result}
            except Exception as e:
                results[step] = {"status": "failed", "error": str(e)}
                break
        
        return results
    
    async def _assess_damage(self):
        health_status = await self.check_all_services()
        return health_status
    
    async def _activate_standby(self):
        await self.k8s_client.scale_deployment("pms-api", 3, namespace="fms-dr")
        return {"standby_activated": True}
    
    async def _restore_data(self):
        latest_backup = await self.get_latest_backup()
        await self.restore_from_backup(latest_backup)
        return {"backup_restored": latest_backup}
    
    async def _verify_services(self):
        health = await self.health_check()
        return health
    
    async def _switch_traffic(self):
        await self.dns_client.update_record(
            "api.fms.example.com",
            self.dr_endpoint
        )
        return {"traffic_switched": True}
    
    async def _notify_stakeholders(self):
        await self.notification_service.send(
            channel="#incidents",
            message="Disaster recovery completed successfully"
        )
        return {"notified": True}
```

class ChaosEngineering:
    def __init__(self, k8s_client):
        self.k8s = k8s_client
    
    async def inject_pod_failure(self, namespace: str, label_selector: str):
        pods = await self.k8s.list_pods(namespace, label_selector)
        target_pod = random.choice(pods)
        
        await self.k8s.delete_pod(namespace, target_pod)
        
        start_time = time.time()
        
        while True:
            await asyncio.sleep(5)
            new_pods = await self.k8s.list_pods(namespace, label_selector)
            
            if len(new_pods) >= len(pods):
                recovery_time = time.time() - start_time
                return {
                    "injected_failure": target_pod,
                    "recovery_time": recovery_time,
                    "status": "recovered"
                }
    
    async def inject_network_latency(self, namespace: str, latency_ms: int):
        network_policy = {
            "apiVersion": "networking.k8s.io/v1",

                "namespace": namespace
            },

                "podSelector": {},
                "policyTypes": ["Ingress"],
                "ingress": [{
                    "from": [{"podSelector": {}}],
                    "withLatency": f"{latency_ms}ms"
                }]
            }
        }
        
        await self.k8s.create_network_policy(network_policy)

## 故障恢复操作手册

#### 3.1 应用层故障

kubectl get pods -n fms-production

# 重启Pod
