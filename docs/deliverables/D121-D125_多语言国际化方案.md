### 2.2 语言检测策略

```python
from enum import Enum
from typing import Optional

class LanguageDetectionStrategy(str, Enum):
    HEADER = "header"
    COOKIE = "cookie"
    URL = "url"
    USER_PREFERENCE = "user_preference"

class LanguageDetector:
    DEFAULT_LANGUAGE = "zh-CN"
    SUPPORTED_LANGUAGES = ["zh-CN", "en-US", "zh-TW", "ja-JP"]
    
    def detect(self, request) -> str:
        language = self._detect_from_url(request)
        if language:
            return language
        
        language = self._detect_from_cookie(request)
        if language:
            return language
        
        language = self._detect_from_header(request)
        if language:
            return language
        
        language = self._detect_from_user_preference(request)
        if language:
            return language
        
        return self.DEFAULT_LANGUAGE
    
    def _detect_from_url(self, request) -> Optional[str]:
        path = request.url.path
        parts = path.split("/")
        
        if len(parts) > 1 and parts[1] in self.SUPPORTED_LANGUAGES:
            return parts[1]
        
        return None
    
    def _detect_from_cookie(self, request) -> Optional[str]:
        language = request.cookies.get("language")
        
        if language in self.SUPPORTED_LANGUAGES:
            return language
        
        return None
    
    def _detect_from_header(self, request) -> Optional[str]:
        accept_language = request.headers.get("Accept-Language", "")
        
        for lang in accept_language.split(","):
            lang = lang.split(";")[0].strip()
            
            if lang in self.SUPPORTED_LANGUAGES:
                return lang
            
            base_lang = lang.split("-")[0]
            for supported in self.SUPPORTED_LANGUAGES:
                if supported.startswith(base_lang):
                    return supported
        
        return None
    
    def _detect_from_user_preference(self, request) -> Optional[str]:
        user = getattr(request.state, "user", None)
        
        if user and hasattr(user, "language"):
            if user.language in self.SUPPORTED_LANGUAGES:
                return user.language
        
        return None
```

## 3. 前端国际化

### 3.1 i18next配置

```javascript
import i18next from 'i18next';
import { initReactI18next } from 'react-i18next';
import LanguageDetector from 'i18next-browser-languagedetector';
import HttpBackend from 'i18next-http-backend';

i18next
  .use(HttpBackend)
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    fallbackLng: 'zh-CN',
    supportedLngs: ['zh-CN', 'en-US', 'zh-TW', 'ja-JP'],
    
    detection: {
      order: ['querystring', 'cookie', 'localStorage', 'navigator'],
      lookupQuerystring: 'lang',
      lookupCookie: 'language',
      lookupLocalStorage: 'language',
      caches: ['localStorage', 'cookie'],
      cookieMinutes: 43200,
      cookieDomain: '.fms.example.com'
    },
    
    backend: {
      loadPath: '/locales/{{lng}}/{{ns}}.json',
      requestInterval: 1000
    },
    
    ns: ['common', 'products', 'selection', 'reports', 'errors'],
    defaultNS: 'common',
    
    interpolation: {
      escapeValue: false
    },
    
    react: {
      useSuspense: true,
      bindI18n: 'languageChanged loaded',
      bindI18nStore: 'added removed'
    }
  });

export default i18next;
```

## 4. 后端国际化

### 4.1 国际化中间件

```python
from fastapi import Request, Response
from starlette.middleware.base import BaseHTTPMiddleware

class I18nMiddleware(BaseHTTPMiddleware):
    def __init__(self, app, default_language: str = "zh-CN"):
        super().__init__(app)
        self.default_language = default_language
        self.detector = LanguageDetector()
    
    async def dispatch(self, request: Request, call_next):
        language = self.detector.detect(request)
        request.state.language = language
        
        response = await call_next(request)
        
        response.set_cookie(
            key="language",
            value=language,
            max_age=43200,
            domain=".fms.example.com"
        )
        
        return response
```

### 4.2 翻译服务

```python
from typing import Optional
import json

class TranslationService:
    def __init__(self, translations_dir: str = "locales"):
        self.translations_dir = translations_dir
        self.cache = {}
    
    def get(
        self,
        key: str,
        language: str = "zh-CN",
        namespace: str = "common",
        **kwargs
    ) -> str:
        translation = self._get_translation(language, namespace, key)
        
        if translation and kwargs:
            return translation.format(**kwargs)
        
        return translation or key
    
    def _get_translation(
        self,
        language: str,
        namespace: str,
        key: str
    ) -> Optional[str]:
        cache_key = f"{language}:{namespace}"
        
        if cache_key not in self.cache:
            self._load_translations(language, namespace)
        
        translations = self.cache.get(cache_key, {})
        
        keys = key.split(".")
        value = translations
        
        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
            else:
                return None
        
        return value if isinstance(value, str) else None
    
    def _load_translations(self, language: str, namespace: str):
        import os
        
        file_path = os.path.join(
            self.translations_dir,
            language,
            f"{namespace}.json"
        )
        
        if os.path.exists(file_path):
            with open(file_path, "r", encoding="utf-8") as f:
                self.cache[f"{language}:{namespace}"] = json.load(f)
        else:
            self.cache[f"{language}:{namespace}"] = {}

def t(key: str, language: str = None, **kwargs) -> str:
    service = get_translation_service()
    
    if language is None:
        language = get_current_language()
    
    return service.get(key, language, **kwargs)
```

### 4.3 错误消息国际化

from enum import Enum

class ErrorCode(str, Enum):
    VALIDATION_ERROR = "validation_error"
    NOT_FOUND = "not_found"
    UNAUTHORIZED = "unauthorized"
    FORBIDDEN = "forbidden"
    INTERNAL_ERROR = "internal_error"

ERROR_MESSAGES = {
    "zh-CN": {
        ErrorCode.VALIDATION_ERROR: "数据验证失败",
        ErrorCode.NOT_FOUND: "资源不存在",
        ErrorCode.UNAUTHORIZED: "未授权访问",
        ErrorCode.FORBIDDEN: "禁止访问",
        ErrorCode.INTERNAL_ERROR: "服务器内部错误"
    },
    "en-US": {
        ErrorCode.VALIDATION_ERROR: "Validation failed",
        ErrorCode.NOT_FOUND: "Resource not found",
        ErrorCode.UNAUTHORIZED: "Unauthorized access",
        ErrorCode.FORBIDDEN: "Access forbidden",
        ErrorCode.INTERNAL_ERROR: "Internal server error"
    }
}

class APIError(Exception):
    def __init__(
        self,
        code: ErrorCode,
        details: dict = None,
        language: str = "zh-CN"
    ):
        self.code = code
        self.message = ERROR_MESSAGES.get(language, {}).get(code, code.value)
        self.details = details or {}
    
    def to_dict(self) -> dict:
        return {
            "error": {
                "code": self.code.value,
                "message": self.message,
                "details": self.details
            }
        }

@app.exception_handler(APIError)
async def api_error_handler(request: Request, exc: APIError):
    language = getattr(request.state, "language", "zh-CN")
    exc.language = language
    
    return JSONResponse(
        status_code=400,
        content=exc.to_dict()
    )

### 5.1 多语言字段设计

```python
from sqlalchemy import Column, String, JSON
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()

class TranslatableMixin:
    def get_translated_field(self, field: str, language: str = "zh-CN") -> str:
        translations = getattr(self, f"{field}_translations", {})
        return translations.get(language) or getattr(self, field, "")

class Product(Base, TranslatableMixin):
    __tablename__ = "products"
    
    product_id = Column(String(36), primary_key=True)
    
    name = Column(String(500))
    name_translations = Column(JSON, default={})
    
    description = Column(String(5000))
    description_translations = Column(JSON, default={})
    
    category = Column(String(100))
    category_translations = Column(JSON, default={})

class Category(Base, TranslatableMixin):
    __tablename__ = "categories"
    
    category_id = Column(String(36), primary_key=True)
    
    name = Column(String(200))
    name_translations = Column(JSON, default={})

class TranslationProvider:
    def translate(
        self,
        text: str,
        source_lang: str,
        target_lang: str
    ) -> str:
        raise NotImplementedError

class DeepLProvider(TranslationProvider):
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.api_url = "https://api.deepl.com/v2/translate"
    
    async def translate(
        self,
        text: str,
        source_lang: str,
        target_lang: str
    ) -> str:
        async with httpx.AsyncClient() as client:
            response = await client.post(
                self.api_url,
                headers={"Authorization": f"DeepL-Auth-Key {self.api_key}"},
                data={
                    "text": text,
                    "source_lang": source_lang.upper(),
                    "target_lang": target_lang.upper()
                }
            )
            
            result = response.json()
            return result["translations"][0]["text"]

class TranslationManager:
    def __init__(self, provider: TranslationProvider):
        self.provider = provider
    
    async def batch_translate(
        self,
        texts: list[str],
        source_lang: str,
        target_lang: str
    ) -> list[str]:
        tasks = [
            self.provider.translate(text, source_lang, target_lang)
            for text in texts
        ]
        
        return await asyncio.gather(*tasks)
```

### 6.3 翻译质量检查

```python
class TranslationQA:
    def __init__(self):
        self.checks = [
            self.check_placeholders,
            self.check_lengths,
            self.check_formatting
        ]
    
    def validate(
        self,
        source: str,
        translation: str
    ) -> dict:
        issues = []
        
        for check in self.checks:
            result = check(source, translation)
            issues.extend(result)
        
        return {
            "valid": len(issues) == 0,
            "issues": issues
        }
    
    def check_placeholders(self, source: str, translation: str) -> list:
        import re
        
        source_placeholders = set(re.findall(r'\{\{.*?\}\}', source))
        translation_placeholders = set(re.findall(r'\{\{.*?\}\}', translation))
        
        issues = []
        
        missing = source_placeholders - translation_placeholders
        if missing:
            issues.append(f"Missing placeholders: {missing}")
        
        extra = translation_placeholders - source_placeholders
        if extra:
            issues.append(f"Extra placeholders: {extra}")
        
        return issues
    
    def check_lengths(self, source: str, translation: str) -> list:
        issues = []
        
        if len(translation) > len(source) * 2:
            issues.append("Translation is significantly longer than source")
        
        return issues
    
    def check_formatting(self, source: str, translation: str) -> list:
        issues = []
        
        if source.endswith('.') and not translation.endswith('.'):
            issues.append("Translation should end with period")
        
        return issues
```
