```python
from pydantic import BaseModel
from datetime import datetime
from typing import Optional
from enum import Enum

class TenantStatus(str, Enum):
    ACTIVE = "active"
    SUSPENDED = "suspended"
    TRIAL = "trial"
    EXPIRED = "expired"

class TenantPlan(str, Enum):
    FREE = "free"
    BASIC = "basic"
    PROFESSIONAL = "professional"
    ENTERPRISE = "enterprise"

class Tenant(BaseModel):
    tenant_id: str
    name: str
    slug: str
    status: TenantStatus = TenantStatus.ACTIVE
    plan: TenantPlan = TenantPlan.BASIC
    
    contact_email: str
    contact_phone: Optional[str] = None
    
    settings: dict = {}
    features: list[str] = []
    
    max_users: int = 10
    max_products: int = 1000
    max_api_calls: int = 10000
    
    created_at: datetime
    updated_at: datetime
    expires_at: Optional[datetime] = None

class TenantContext:
    _current_tenant: Optional[Tenant] = None
    
    @classmethod
    def set_tenant(cls, tenant: Tenant):
        cls._current_tenant = tenant
    
    @classmethod
    def get_tenant(cls) -> Optional[Tenant]:
        return cls._current_tenant
    
    @classmethod
    def get_tenant_id(cls) -> Optional[str]:
        return cls._current_tenant.tenant_id if cls._current_tenant else None
```

### 2.3 数据隔离实现

```python
from sqlalchemy import Column, String, ForeignKey
from sqlalchemy.orm import declared_attr
from sqlalchemy.ext.declarative import declared_attr

class TenantMixin:
    @declared_attr
    def tenant_id(cls):
        return Column(String(36), ForeignKey("tenants.tenant_id"), index=True)

class TenantQuery:
    def __init__(self, model):
        self.model = model
    
    def filter_by_tenant(self, query):
        tenant_id = TenantContext.get_tenant_id()
        if tenant_id:
            query = query.filter(self.model.tenant_id == tenant_id)
        return query

class TenantService:
    def __init__(self, db: Database):
        self.db = db
    
    async def create_tenant(self, tenant: Tenant) -> Tenant:
        tenant.tenant_id = str(uuid.uuid4())
        tenant.created_at = datetime.now()
        tenant.updated_at = datetime.now()
        
        await self.db.insert("tenants", tenant.dict())
        return tenant
    
    async def get_tenant(self, tenant_id: str) -> Optional[Tenant]:
        data = await self.db.get("tenants", tenant_id)
        if data:
            return Tenant(**data)
        return None
    
    async def update_tenant(self, tenant_id: str, updates: dict) -> bool:
        updates["updated_at"] = datetime.now().isoformat()
        return await self.db.update("tenants", {"tenant_id": tenant_id}, updates)
    
    async def suspend_tenant(self, tenant_id: str) -> bool:
        return await self.update_tenant(tenant_id, {"status": TenantStatus.SUSPENDED})
```

## 3. 权限系统设计

### 3.1 RBAC模型

```python
from enum import Enum
from typing import Optional

class Permission(str, Enum):
    READ = "read"
    WRITE = "write"
    DELETE = "delete"
    ADMIN = "admin"
    EXPORT = "export"
    IMPORT = "import"

class Role(str, Enum):
    SUPER_ADMIN = "super_admin"
    TENANT_ADMIN = "tenant_admin"
    MANAGER = "manager"
    ANALYST = "analyst"
    OPERATOR = "operator"
    VIEWER = "viewer"

ROLE_PERMISSIONS = {
    Role.SUPER_ADMIN: [p for p in Permission],
    Role.TENANT_ADMIN: [
        Permission.READ, Permission.WRITE, Permission.DELETE,
        Permission.ADMIN, Permission.EXPORT, Permission.IMPORT
    ],
    Role.MANAGER: [
        Permission.READ, Permission.WRITE, Permission.DELETE,
        Permission.EXPORT, Permission.IMPORT
    ],
    Role.ANALYST: [
        Permission.READ, Permission.WRITE, Permission.EXPORT
    ],
    Role.OPERATOR: [
        Permission.READ, Permission.WRITE
    ],
    Role.VIEWER: [
        Permission.READ
    ]
}

class User(BaseModel):
    user_id: str
    tenant_id: str
    username: str
    email: str
    roles: list[Role]
    permissions: list[Permission] = []
    is_active: bool = True
    created_at: datetime

class RoleService:
    def __init__(self, db: Database):
        self.db = db
    
    async def assign_role(self, user_id: str, role: Role) -> bool:
        user = await self.get_user(user_id)
        if role not in user.roles:
            user.roles.append(role)
            await self.update_user(user)
        return True
    
    async def revoke_role(self, user_id: str, role: Role) -> bool:
        user = await self.get_user(user_id)
        if role in user.roles:
            user.roles.remove(role)
            await self.update_user(user)
        return True
    
    def get_permissions(self, roles: list[Role]) -> list[Permission]:
        permissions = set()
        for role in roles:
            permissions.update(ROLE_PERMISSIONS.get(role, []))
        return list(permissions)
    
    def has_permission(self, user: User, permission: Permission) -> bool:
        return permission in user.permissions
```

### 3.2 资源权限控制

```python
from functools import wraps

class ResourcePermission:
    def __init__(self, resource: str, action: Permission):
        self.resource = resource
        self.action = action

def require_permission(resource: str, action: Permission):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, current_user: User = None, **kwargs):
            if not current_user:
                raise AuthenticationError("Not authenticated")
            
            if not PermissionService.has_permission(current_user, action):
                raise AuthorizationError(f"No permission to {action} {resource}")
            
            return await func(*args, current_user=current_user, **kwargs)
        return wrapper
    return decorator

class PermissionService:
    @staticmethod
    def has_permission(user: User, permission: Permission) -> bool:
        return permission in user.permissions
    
    @staticmethod
    def check_resource_access(
        user: User,
        resource_type: str,
        resource_id: str,
        action: Permission
    ) -> bool:
        if Permission.ADMIN in user.permissions:
            return True
        
        if not PermissionService.has_permission(user, action):
            return False
        
        return True
```

### 4.1 CRM数据模型

```python
class Customer(BaseModel):
    customer_id: str
    tenant_id: str
    name: str
    email: str
    phone: Optional[str] = None
    company: Optional[str] = None
    
    source: str
    status: str = "lead"
    score: int = 0
    
    tags: list[str] = []
    custom_fields: dict = {}
    
    created_at: datetime
    updated_at: datetime

class CustomerFeedback(BaseModel):
    feedback_id: str
    customer_id: str
    tenant_id: str
    product_id: str
    
    rating: int
    title: str
    content: str
    
    sentiment: str = "neutral"
    categories: list[str] = []
    
    status: str = "pending"
    response: Optional[str] = None
    
    created_at: datetime

class CRMService:
    def __init__(self, db: Database):
        self.db = db
    
    async def create_customer(self, customer: Customer) -> Customer:
        customer.customer_id = str(uuid.uuid4())
        customer.created_at = datetime.now()
        customer.updated_at = datetime.now()
        
        await self.db.insert("customers", customer.dict())
        return customer
    
    async def get_customers(
        self,
        tenant_id: str,
        status: str = None,
        page: int = 1,
        size: int = 20
    ) -> list[Customer]:
        query = {"tenant_id": tenant_id}
        if status:
            query["status"] = status
        
        customers = await self.db.query(
            "customers",
            query,
            offset=(page - 1) * size,
            limit=size
        )
        return [Customer(**c) for c in customers]
    
    async def update_customer_score(
        self,
        customer_id: str,
        score_delta: int
    ) -> bool:
        customer = await self.get_customer(customer_id)
        new_score = max(0, min(100, customer.score + score_delta))
        
        return await self.db.update(
            "customers",
            {"customer_id": customer_id},
            {"score": new_score, "updated_at": datetime.now().isoformat()}
        )
```

### 5.1 FMS数据模型

```python
from decimal import Decimal

class FinancialRecord(BaseModel):
    record_id: str
    tenant_id: str
    record_type: str
    
    amount: Decimal
    currency: str = "USD"
    
    category: str
    subcategory: Optional[str] = None
    
    reference_id: Optional[str] = None
    reference_type: Optional[str] = None
    
    description: str
    transaction_date: datetime
    
    status: str = "pending"
    
    created_at: datetime

class CostCenter(BaseModel):
    center_id: str
    tenant_id: str
    name: str
    code: str
    
    budget: Decimal
    spent: Decimal = Decimal("0")
    
    manager_id: str
    
    created_at: datetime

class FMSService:
    def __init__(self, db: Database):
        self.db = db
    
    async def create_record(self, record: FinancialRecord) -> FinancialRecord:
        record.record_id = str(uuid.uuid4())
        record.created_at = datetime.now()
        
        await self.db.insert("financial_records", record.dict())
        return record
    
    async def get_financial_summary(
        self,
        tenant_id: str,
        start_date: datetime,
        end_date: datetime
    ) -> dict:
        records = await self.db.query(
            "financial_records",
            {
                "tenant_id": tenant_id,
                "transaction_date": {
                    "$gte": start_date.isoformat(),
                    "$lte": end_date.isoformat()
                }
            }
        )
        
        summary = {
            "total_revenue": Decimal("0"),
            "total_expense": Decimal("0"),
            "by_category": {}
        }
        
        for record in records:
            amount = Decimal(record["amount"])
            if record["record_type"] == "revenue":
                summary["total_revenue"] += amount
            else:
                summary["total_expense"] += amount
            
            category = record["category"]
            if category not in summary["by_category"]:
                summary["by_category"][category] = Decimal("0")
            summary["by_category"][category] += amount
        
        return summary
    
    async def calculate_product_cost(
        self,
        tenant_id: str,
        product_id: str
    ) -> dict:
        costs = await self.db.query(
            "financial_records",
            {
                "tenant_id": tenant_id,
                "reference_type": "product",
                "reference_id": product_id
            }
        )
        
        total_cost = sum(Decimal(c["amount"]) for c in costs)
        
        return {
            "product_id": product_id,
            "total_cost": total_cost,
            "cost_breakdown": costs
        }
```
