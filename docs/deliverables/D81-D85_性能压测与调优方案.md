## 1. 概述

### 1.1 压测目标

通过系统化的性能压测，验证系统在高并发场景下的稳定性，识别性能瓶颈并进行优化。

### 1.2 压测范围

| 模块 | 压测场景 | 优先级 |
|------|---------|--------|
| API服务 | 并发请求、响应时间 | P0 |
| Agent服务 | 任务处理吞吐量 | P0 |
| LLM推理 | 推理延迟、吞吐量 | P0 |
| 数据库 | 查询性能、连接池 | P1 |
| 缓存 | 命中率、响应时间 | P1 |

### 4.1 Locust压测脚本

```python
from locust import HttpUser, task, between, events
from locust.runners import MasterRunner, WorkerRunner
import json
import random

class FMSUser(HttpUser):
    wait_time = between(1, 3)
    
    def on_start(self):
        self.login()
        self.products = self.get_products()
    
    def login(self):
        response = self.client.post("/api/v1/auth/login", json={
            "username": "test_user",
            "password": "test_password"
        })
        if response.status_code == 200:
            self.token = response.json().get("access_token")
            self.headers = {"Authorization": f"Bearer {self.token}"}
    
    def get_products(self):
        response = self.client.get(
            "/api/v1/products",
            headers=self.headers,
            params={"page": 1, "size": 100}
        )
        if response.status_code == 200:
            return response.json().get("products", [])
        return []
    
    @task(10)
    def search_products(self):
        keywords = ["户外", "储能", "电源", "露营", "便携"]
        keyword = random.choice(keywords)
        
        self.client.get(
            "/api/v1/products/search",
            headers=self.headers,
            params={"keyword": keyword, "page": 1, "size": 20},
            name="/products/search"
        )
    
    @task(5)
    def get_product_detail(self):
        if not self.products:
            return
        
        product = random.choice(self.products)
        product_id = product.get("product_id")
        
        self.client.get(
            f"/api/v1/products/{product_id}",
            headers=self.headers,
            name="/products/[id]"
        )
    
    @task(3)
    def analyze_market(self):
        categories = ["户外用品", "电子产品", "家居用品"]
        category = random.choice(categories)
        
        self.client.post(
            "/api/v1/market-insight/analyze",
            headers=self.headers,
            json={"category": category},
            name="/market-insight/analyze"
        )
    
    @task(2)
    def create_selection_task(self):
        self.client.post(
            "/api/v1/selection/tasks",
            headers=self.headers,
            json={
                "category": "户外用品",
                "budget": 10000,
                "target_market": "US"
            },
            name="/selection/tasks"
        )
    
    @task(1)
    def get_agent_status(self):
        self.client.get(
            "/api/v1/agents/status",
            headers=self.headers,
            name="/agents/status"
        )

@events.test_start.add_listener
def on_test_start(environment, **kwargs):
    if isinstance(environment.runner, MasterRunner):
        print("Starting performance test...")

@events.test_stop.add_listener
def on_test_stop(environment, **kwargs):
    if isinstance(environment.runner, MasterRunner):
        print("Performance test completed.")
```

import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate, Trend, Counter } from 'k6/metrics';

const errorRate = new Rate('errors');
const latencyTrend = new Trend('latency');
const requestCounter = new Counter('requests');

export const options = {
    stages: [
        { duration: '2m', target: 100 },
        { duration: '5m', target: 100 },
        { duration: '2m', target: 200 },
        { duration: '5m', target: 200 },
        { duration: '2m', target: 500 },
        { duration: '5m', target: 500 },
        { duration: '2m', target: 1000 },
        { duration: '5m', target: 1000 },
        { duration: '5m', target: 0 },
    ],
    thresholds: {
        http_req_duration: ['p(99)<200', 'p(95)<100'],
        errors: ['rate<0.01'],
    },
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:8000';

export default function () {
    const loginRes = http.post(`${BASE_URL}/api/v1/auth/login`, JSON.stringify({
        username: 'test_user',
        password: 'test_password',
    }), {
        headers: { 'Content-Type': 'application/json' },
    });

    check(loginRes, {
        'login successful': (r) => r.status === 200,
    });

    const token = loginRes.json('access_token');
    const headers = {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json',
    };

    const searchRes = http.get(`${BASE_URL}/api/v1/products/search?keyword=户外`, {
        headers: headers,
    });

    check(searchRes, {
        'search successful': (r) => r.status === 200,
        'search latency < 100ms': (r) => r.timings.duration < 100,
    });

    errorRate.add(searchRes.status !== 200);
    latencyTrend.add(searchRes.timings.duration);
    requestCounter.add(1);

    sleep(1);
}

### 4.3 数据库压测脚本

```python
import asyncio
import time
from sqlalchemy import text
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession

class DatabaseBenchmark:
    def __init__(self, db_url: str):
        self.engine = create_async_engine(db_url, pool_size=50, max_overflow=10)
    
    async def run_benchmark(self, queries: int, concurrency: int):
        start_time = time.time()
        
        tasks = []
        for _ in range(queries):
            if len(tasks) >= concurrency:
                await asyncio.gather(*tasks)
                tasks = []
            tasks.append(self._execute_query())
        
        if tasks:
            await asyncio.gather(*tasks)
        
        duration = time.time() - start_time
        qps = queries / duration
        
        print(f"Total queries: {queries}")
        print(f"Duration: {duration:.2f}s")
        print(f"QPS: {qps:.2f}")
    
    async def _execute_query(self):
        async with AsyncSession(self.engine) as session:
            result = await session.execute(
                text("SELECT * FROM products WHERE category = :category LIMIT 10"),
                {"category": "户外用品"}
            )
            return result.fetchall()
    
    async def close(self):
        await self.engine.dispose()

async def main():
    benchmark = DatabaseBenchmark("postgresql+asyncpg://user:pass@localhost/fms")
    await benchmark.run_benchmark(queries=10000, concurrency=100)
    await benchmark.close()

if __name__ == "__main__":
    asyncio.run(main())
```

### 5.3 缓存调优

```python
class CacheStrategy:
    def __init__(self, redis_client):
        self.redis = redis_client
        self.local_cache = TTLCache(maxsize=10000, ttl=60)
    
    async def get_with_cache(
        self,
        key: str,
        fetch_func: callable,
        ttl: int = 300
    ):
        if key in self.local_cache:
            return self.local_cache[key]
        
        cached = await self.redis.get(key)
        if cached:
            self.local_cache[key] = cached
            return cached
        
        data = await fetch_func()
        
        await self.redis.setex(key, ttl, data)
        self.local_cache[key] = data
        
        return data
    
    async def invalidate(self, key: str):
        self.local_cache.pop(key, None)
        await self.redis.delete(key)

class CacheWarmup:
    def __init__(self, cache: CacheStrategy):
        self.cache = cache
    
    async def warmup_hot_products(self):
        hot_products = await self._get_hot_products()
        
        tasks = []
        for product in hot_products:
            key = f"product:{product['id']}"
            tasks.append(self.cache.redis.setex(key, 3600, product))
        
        await asyncio.gather(*tasks)
    
    async def warmup_categories(self):
        categories = await self._get_categories()
        
        tasks = []
        for category in categories:
            key = f"category:{category['id']}"
            tasks.append(self.cache.redis.setex(key, 7200, category))
        
        await asyncio.gather(*tasks)
```

from sqlalchemy.ext.asyncio import create_async_engine
from sqlalchemy.pool import QueuePool

class DatabasePool:
    def __init__(self, db_url: str):
        self.engine = create_async_engine(
            db_url,
            poolclass=QueuePool,
            pool_size=20,
            max_overflow=10,
            pool_timeout=30,
            pool_recycle=1800,
            pool_pre_ping=True,
            echo=False
        )
    
    async def get_connection(self):
        async with self.engine.connect() as conn:
            yield conn
    
    async def close(self):
        await self.engine.dispose()

### 6.1 容量模型

```python
class CapacityPlanner:
    def __init__(self):
        self.metrics = {}
    
    def calculate_capacity(self, target_qps: int, avg_latency_ms: float) -> dict:
        concurrent_requests = target_qps * (avg_latency_ms / 1000)
        
        cpu_per_request = 0.001
        memory_per_request = 10 * 1024 * 1024
        
        required_cpu = concurrent_requests * cpu_per_request
        required_memory = concurrent_requests * memory_per_request
        
        cpu_cores = max(4, int(required_cpu * 1.5))
        memory_gb = max(8, int(required_memory / (1024**3) * 1.5))
        
        return {
            "target_qps": target_qps,
            "avg_latency_ms": avg_latency_ms,
            "concurrent_requests": concurrent_requests,
            "recommended_cpu_cores": cpu_cores,
            "recommended_memory_gb": memory_gb,
            "recommended_instances": max(3, int(cpu_cores / 8))
        }
    
    def estimate_storage(self, data_growth_rate: float, retention_days: int) -> dict:
        daily_growth = data_growth_rate
        total_storage = daily_growth * retention_days
        
        return {
            "daily_growth_gb": daily_growth,
            "retention_days": retention_days,
            "total_storage_gb": total_storage,
            "recommended_storage_gb": total_storage * 1.5
        }
```
