### 2.2 数据模型

from pydantic import BaseModel
from datetime import datetime
from typing import Optional, Any
from enum import Enum

class MetricType(str, Enum):
    COUNT = "count"
    SUM = "sum"
    AVG = "avg"
    MAX = "max"
    MIN = "min"
    PERCENTILE = "percentile"

class Dimension(BaseModel):
    name: str
    field: str
    type: str = "string"
    values: list[Any] = []

class Metric(BaseModel):
    name: str
    field: str
    type: MetricType
    aggregation: str = "sum"

class AnalysisQuery(BaseModel):
    query_id: str
    name: str
    description: str
    
    dimensions: list[Dimension]
    metrics: list[Metric]
    
    filters: list[dict] = []
    time_range: dict = {}
    
    limit: int = 1000
    offset: int = 0

class AnalysisResult(BaseModel):
    query_id: str
    dimensions: list[str]
    metrics: list[str]
    
    data: list[dict]
    total: int
    
    execution_time: float
    cached: bool = False

### 2.3 查询引擎

```python
class QueryEngine:
    def __init__(self, db: Database, cache: RedisCache):
        self.db = db
        self.cache = cache
    
    async def execute(self, query: AnalysisQuery) -> AnalysisResult:
        cache_key = self._get_cache_key(query)
        
        cached = await self.cache.get(cache_key)
        if cached:
            return AnalysisResult(**cached, cached=True)
        
        start_time = time.time()
        
        sql = self._build_sql(query)
        data = await self.db.execute(sql)
        
        execution_time = time.time() - start_time
        
        result = AnalysisResult(
            query_id=query.query_id,
            dimensions=[d.name for d in query.dimensions],
            metrics=[m.name for m in query.metrics],
            data=data,
            total=len(data),
            execution_time=execution_time
        )
        
        await self.cache.set(cache_key, result.dict(), ttl=300)
        
        return result
    
    def _build_sql(self, query: AnalysisQuery) -> str:
        select_fields = []
        
        for dim in query.dimensions:
            select_fields.append(dim.field)
        
        for metric in query.metrics:
            if metric.type == MetricType.COUNT:
                select_fields.append(f"COUNT({metric.field}) as {metric.name}")
            elif metric.type == MetricType.SUM:
                select_fields.append(f"SUM({metric.field}) as {metric.name}")
            elif metric.type == MetricType.AVG:
                select_fields.append(f"AVG({metric.field}) as {metric.name}")
        
        group_by = [d.field for d in query.dimensions]
        
        sql = f"""
        SELECT {', '.join(select_fields)}
        FROM analysis_view
        WHERE 1=1
        """
        
        if query.time_range:
            sql += f" AND created_at BETWEEN '{query.time_range['start']}' AND '{query.time_range['end']}'"
        
        for f in query.filters:
            sql += f" AND {f['field']} {f['operator']} {f['value']}"
        
        if group_by:
            sql += f" GROUP BY {', '.join(group_by)}"
        
        sql += f" LIMIT {query.limit} OFFSET {query.offset}"
        
        return sql
    
    def _get_cache_key(self, query: AnalysisQuery) -> str:
        import hashlib
        import json
        
        content = json.dumps(query.dict(), sort_keys=True)
        return f"analysis:{hashlib.md5(content.encode()).hexdigest()}"
```

### 3.1 报表模板

```python
class ReportTemplate(BaseModel):
    template_id: str
    name: str
    description: str
    category: str
    
    layout: dict
    components: list[dict]
    data_sources: list[dict]
    
    parameters: list[dict] = []
    
    created_at: datetime
    updated_at: datetime

class Report(BaseModel):
    report_id: str
    template_id: str
    name: str
    
    parameters: dict = {}
    data: dict = {}
    
    status: str = "pending"
    created_at: datetime
    generated_at: Optional[datetime] = None

class ReportEngine:
    def __init__(self, db: Database, query_engine: QueryEngine):
        self.db = db
        self.query_engine = query_engine
        self.templates = {}
    
    async def generate_report(
        self,
        template_id: str,
        parameters: dict = {}
    ) -> Report:
        template = await self.get_template(template_id)
        
        report = Report(
            report_id=str(uuid.uuid4()),
            template_id=template_id,
            name=template.name,
            parameters=parameters,
            status="generating"
        )
        
        data = {}
        for source in template.data_sources:
            query = AnalysisQuery(**source["query"])
            for key, value in parameters.items():
                self._apply_parameter(query, key, value)
            
            result = await self.query_engine.execute(query)
            data[source["name"]] = result.data
        
        report.data = data
        report.status = "completed"
        report.generated_at = datetime.now()
        
        return report
    
    async def get_template(self, template_id: str) -> ReportTemplate:
        if template_id in self.templates:
            return self.templates[template_id]
        
        template_data = await self.db.get("report_templates", template_id)
        template = ReportTemplate(**template_data)
        self.templates[template_id] = template
        
        return template
```

## 4. 数据可视化

### 4.1 图表类型

```python
from enum import Enum

class ChartType(str, Enum):
    LINE = "line"
    BAR = "bar"
    PIE = "pie"
    AREA = "area"
    SCATTER = "scatter"
    RADAR = "radar"
    GAUGE = "gauge"
    HEATMAP = "heatmap"
    TREEMAP = "treemap"
    FUNNEL = "funnel"
    SANKEY = "sankey"
    WORDCLOUD = "wordcloud"
    MAP = "map"
    TABLE = "table"
    KPI = "kpi"

class ChartConfig(BaseModel):
    chart_type: ChartType
    title: str
    subtitle: str = ""
    
    x_axis: Optional[dict] = None
    y_axis: Optional[dict] = None
    
    series: list[dict]
    
    legend: dict = {"show": True}
    tooltip: dict = {"show": True}
    
    colors: list[str] = []
    
    width: int = 800
    height: int = 400

class ChartRenderer:
    def __init__(self):
        self.renderers = {
            ChartType.LINE: self._render_line,
            ChartType.BAR: self._render_bar,
            ChartType.PIE: self._render_pie,
            ChartType.TABLE: self._render_table,
            ChartType.KPI: self._render_kpi
        }
    
    def render(self, config: ChartConfig, data: list[dict]) -> dict:
        renderer = self.renderers.get(config.chart_type)
        if renderer:
            return renderer(config, data)
        return {}
    
    def _render_line(self, config: ChartConfig, data: list[dict]) -> dict:
        return {
            "type": "line",
            "title": config.title,
            "xAxis": {
                "data": [d[config.x_axis["field"]] for d in data]
            },
            "yAxis": {},
            "series": [
                {
                    "name": s["name"],
                    "type": "line",
                    "data": [d[s["field"]] for d in data]
                }
                for s in config.series
            ]
        }
    
    def _render_bar(self, config: ChartConfig, data: list[dict]) -> dict:
        return {
            "type": "bar",
            "title": config.title,
            "xAxis": {
                "data": [d[config.x_axis["field"]] for d in data]
            },
            "yAxis": {},
            "series": [
                {
                    "name": s["name"],
                    "type": "bar",
                    "data": [d[s["field"]] for d in data]
                }
                for s in config.series
            ]
        }
    
    def _render_pie(self, config: ChartConfig, data: list[dict]) -> dict:
        return {
            "type": "pie",
            "title": config.title,
            "series": [
                {
                    "name": config.series[0]["name"],
                    "type": "pie",
                    "data": [
                        {"name": d[config.series[0]["category"]], "value": d[config.series[0]["field"]]}
                        for d in data
                    ]
                }
            ]
        }
    
    def _render_table(self, config: ChartConfig, data: list[dict]) -> dict:
        columns = [s["name"] for s in config.series]
        rows = [[d.get(s["field"], "") for s in config.series] for d in data]
        
        return {
            "type": "table",
            "title": config.title,
            "columns": columns,
            "rows": rows
        }
    
    def _render_kpi(self, config: ChartConfig, data: list[dict]) -> dict:
        value = data[0].get(config.series[0]["field"], 0) if data else 0
        
        return {
            "type": "kpi",
            "title": config.title,
            "value": value,
            "unit": config.series[0].get("unit", "")
        }
```

### 4.2 Dashboard设计

```python
class Dashboard(BaseModel):
    dashboard_id: str
    name: str
    description: str
    
    layout: list[dict]
    widgets: list[dict]
    
    filters: list[dict] = []
    
    created_at: datetime
    updated_at: datetime

class DashboardWidget(BaseModel):
    widget_id: str
    dashboard_id: str
    
    title: str
    chart_type: ChartType
    
    data_source: dict
    refresh_interval: int = 300
    
    position: dict
    size: dict

class DashboardService:
    def __init__(self, db: Database, query_engine: QueryEngine):
        self.db = db
        self.query_engine = query_engine
        self.renderer = ChartRenderer()
    
    async def get_dashboard(self, dashboard_id: str) -> Dashboard:
        data = await self.db.get("dashboards", dashboard_id)
        return Dashboard(**data)
    
    async def render_dashboard(
        self,
        dashboard_id: str,
        filters: dict = {}
    ) -> dict:
        dashboard = await self.get_dashboard(dashboard_id)
        
        widgets_data = []
        for widget_config in dashboard.widgets:
            widget = DashboardWidget(**widget_config)
            
            query = AnalysisQuery(**widget.data_source)
            for key, value in filters.items():
                self._apply_filter(query, key, value)
            
            result = await self.query_engine.execute(query)
            
            chart_config = ChartConfig(
                chart_type=widget.chart_type,
                title=widget.title,
                series=widget.data_source.get("series", [])
            )
            
            rendered = self.renderer.render(chart_config, result.data)
            
            widgets_data.append({
                "widget_id": widget.widget_id,
                "position": widget.position,
                "size": widget.size,
                "chart": rendered
            })
        
        return {
            "dashboard_id": dashboard_id,
            "name": dashboard.name,
            "widgets": widgets_data
        }
```

## 5. 数据导出功能

### 5.1 导出格式

```python
from io import BytesIO
import pandas as pd
from reportlab.lib import colors
from reportlab.lib.pagesizes import letter
from reportlab.platypus import SimpleDocTemplate, Table, TableStyle

class ExportFormat(str, Enum):
    CSV = "csv"
    EXCEL = "excel"
    PDF = "pdf"
    JSON = "json"

class DataExporter:
    def __init__(self):
        self.exporters = {
            ExportFormat.CSV: self._export_csv,
            ExportFormat.EXCEL: self._export_excel,
            ExportFormat.PDF: self._export_pdf,
            ExportFormat.JSON: self._export_json
        }
    
    async def export(
        self,
        data: list[dict],
        format: ExportFormat,
        options: dict = {}
    ) -> BytesIO:
        exporter = self.exporters.get(format)
        if exporter:
            return await exporter(data, options)
        raise ValueError(f"Unsupported format: {format}")
    
    async def _export_csv(self, data: list[dict], options: dict) -> BytesIO:
        df = pd.DataFrame(data)
        
        buffer = BytesIO()
        df.to_csv(buffer, index=False, encoding="utf-8-sig")
        buffer.seek(0)
        
        return buffer
    
    async def _export_excel(self, data: list[dict], options: dict) -> BytesIO:
        df = pd.DataFrame(data)
        
        buffer = BytesIO()
        
        with pd.ExcelWriter(buffer, engine="openpyxl") as writer:
            df.to_excel(writer, index=False, sheet_name="Sheet1")
        
        buffer.seek(0)
        return buffer
    
    async def _export_pdf(self, data: list[dict], options: dict) -> BytesIO:
        buffer = BytesIO()
        
        doc = SimpleDocTemplate(buffer, pagesize=letter)
        elements = []
        
        if data:
            df = pd.DataFrame(data)
            table_data = [df.columns.tolist()] + df.values.tolist()
            
            table = Table(table_data)
            table.setStyle(TableStyle([
                ('BACKGROUND', (0, 0), (-1, 0), colors.grey),
                ('TEXTCOLOR', (0, 0), (-1, 0), colors.whitesmoke),
                ('ALIGN', (0, 0), (-1, -1), 'CENTER'),
                ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
                ('FONTSIZE', (0, 0), (-1, 0), 12),
                ('BOTTOMPADDING', (0, 0), (-1, 0), 12),
                ('GRID', (0, 0), (-1, -1), 1, colors.black)
            ]))
            
            elements.append(table)
        
        doc.build(elements)
        buffer.seek(0)
        
        return buffer
    
    async def _export_json(self, data: list[dict], options: dict) -> BytesIO:
        buffer = BytesIO()
        buffer.write(json.dumps(data, ensure_ascii=False, indent=2).encode())
        buffer.seek(0)
        
        return buffer
```
