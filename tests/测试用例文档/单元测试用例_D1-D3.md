### 4.2 自动化测试脚本

```python
class TestD3_K8sMasterDeployment:
    """
    D3 K8s Master节点部署验证测试。
    
    注意: 此类测试需要实际K8s集群环境，
    在无集群环境时跳过。
    """
    
    @pytest.mark.skipif(
        not os.environ.get("KUBECONFIG"),
        reason="需要K8s集群访问权限"
    )
    def test_kubectl_configured(self):
        """验证kubectl已正确配置。"""
        import subprocess
        result = subprocess.run(
            ["kubectl", "get", "nodes"],
            capture_output=True,
            text=True
        )
        assert result.returncode == 0, f"kubectl配置失败: {result.stderr}"
    
    @pytest.mark.skipif(
        not os.environ.get("KUBECONFIG"),
        reason="需要K8s集群访问权限"
    )
    def test_nodes_ready(self):
        """验证所有节点状态为Ready。"""
        import subprocess
        result = subprocess.run(
            ["kubectl", "get", "nodes", "-o", "jsonpath={.items[*].status.conditions[?(@.type=='Ready')].status}"],
            capture_output=True,
            text=True
        )
        statuses = result.stdout.strip().split()
        assert all(s == "True" for s in statuses), "存在非Ready节点"
    
    @pytest.mark.skipif(
        not os.environ.get("KUBECONFIG"),
        reason="需要K8s集群访问权限"
    )
    def test_system_pods_running(self):
        """验证系统命名空间Pod正常运行。"""
        import subprocess
        result = subprocess.run(
            ["kubectl", "get", "pods", "-n", "kube-system", "-o", "jsonpath={.items[*].status.phase}"],
            capture_output=True,
            text=True
        )
        phases = result.stdout.strip().split()
        assert all(p == "Running" for p in phases), "存在非Running的系统Pod"
    
    @pytest.mark.skipif(
        not os.environ.get("KUBECONFIG"),
        reason="需要K8s集群访问权限"
    )
    def test_coredns_available(self):
        """验证CoreDNS服务可用。"""
        import subprocess
        result = subprocess.run(
            ["kubectl", "get", "pods", "-n", "kube-system", "-l", "k8s-app=kube-dns"],
            capture_output=True,
            text=True
        )
        assert "Running" in result.stdout, "CoreDNS Pod未运行"
    
    @pytest.mark.skipif(
        not os.environ.get("KUBECONFIG"),
        reason="需要K8s集群访问权限"
    )
    def test_cluster_dns_resolution(self):
        """验证集群DNS解析正常。"""
        import subprocess
        # 创建测试Pod并执行DNS查询
        result = subprocess.run(
            ["kubectl", "run", "dns-test", "--image=busybox", "--rm", "-it", "--restart=Never", "--", "nslookup", "kubernetes.default"],
            capture_output=True,
            text=True,
            timeout=30
        )
        # 检查输出是否包含IP地址
        assert "Address" in result.stdout or "address" in result.stdout.lower()
```
