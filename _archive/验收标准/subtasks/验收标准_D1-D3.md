### 关键验证脚本
```bash
#!/bin/bash
echo "=== D3 Master节点验收 ==="
echo "1. 节点状态:"
kubectl get nodes -o wide
echo "2. 组件状态:"
kubectl get cs 2>/dev/null || kubectl get componentstatuses
echo "3. System Pods:"
kubectl get pods -A | grep -v Running | head -5
echo "4. DNS测试:"
kubectl run dns-test --image=busybox --rm -it -- nslookup kubernetes.default.svc.cluster.local 2>&1 | tail -3
echo "=== 验收完成 ==="
```
