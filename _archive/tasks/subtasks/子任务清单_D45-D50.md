async def trace_middleware(request: Request, call_next):
    trace_id = request.headers.get("X-Trace-ID", generate_trace_id())
    # 注入到所有下游调用
    response = await call_next(request)
    response.headers["X-Trace-ID"] = trace_id
    return response
